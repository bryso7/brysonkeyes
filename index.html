<body>

    <textarea id="textField" placeholder="Start typing..." autofocus></textarea>

    <button id="clearBtn">Clear</button>

    

    <script>

        const textField = document.getElementById('textField');

        const clearBtn = document.getElementById('clearBtn');

        

        // Load saved text from memory

        const savedText = sessionStorage.getItem('textContent');

        if (savedText) {

            textField.value = savedText;

        }

        

        // Save text as user types

        textField.addEventListener('input', () => {

            sessionStorage.setItem('textContent', textField.value);

        });

        

        // Clear button functionality

        clearBtn.addEventListener('click', () => {

            textField.value = '';

            sessionStorage.removeItem('textContent');

            textField.focus();

        });

    </script>

</body>
